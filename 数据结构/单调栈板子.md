单调栈板子

```c++
#include<iostream>
using namespace std;
const int N=3e6+10;
int s[N];
int ans[N];
int a[N];
int r=0;
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
        {
            cin>>a[i];
        }
    for(int i=1;i<=n;i++)
        {
            while(r>0&&a[s[r-1]]<a[i])//求最右边大于它的第一个元素，遇到比它大的弹出
                {
                    int cur=s[--r];
                    ans[cur]=i;
                }
            s[r++]=i;
        }
    while(r>0)
        {
            int cur=s[--r];
            ans[cur]=0;
        }
    for(int i=1;i<=n;i++)
    cout<<ans[i]<<" ";
    return 0;
}
```
